JS

                            
                                <script>
                                    $('.quantity-arrow-plus').click(function () {
                                        console.log($(this));
                                        if ($(this).prev().val() < 5) {

                                            const qv = $(this).prev().val();

                                            $(this).prev().val(parseInt(qv) + 1);
                                        }
                                    });
                                    $('.quantity-arrow-minus').click(function () {
                                        if ($(this).next().val() > 1) {
                                            if ($(this).next().val() > 1) $(this).next().val(+$(this).next().val() - 1);
                                        }
                                    });
                                </script>
